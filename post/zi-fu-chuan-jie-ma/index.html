字符串解码

思路：栈操作

这道题目可以用两个栈来实现，一个栈存储数字，另一个栈用来存储字符串

代码实现

class Solution {
private:
    stack<int> stNum;
    stack<string> stStr;
public:
    string decodeString(string s) {
        int num = 0;
        string str = "";
        for(int i=0;i<s.size();){
            if(isdigit(s[i])){
                num = num*10 + s[i] - '0';
                i++;
            }else if(s[i] == '['){  //入栈
                if(num == 0){
                    //特殊处理
                    num = 1;
                }   
                stNum.push(num);
                stStr.push("[");
                num = 0;
                i++;
            }else if(s[i] == ']'){
                //出栈
                int count = stNum.top();
                stNum.pop();
                string s,temp;
                while(!stStr.empty() && stStr.top()!="["){
                    temp = stStr.top() + temp;
                    stStr.pop();
                }
                for(int i=0;i<count;i++){
                    s += temp;
                }
                if(stStr.top()== "["){
                    stStr.pop();
                }
                stStr.push(s);
                i++; 
            }else{
                while(isalpha(s[i])){
                    str += s[i++];
                }
                stStr.push(str);
                str = "";
            }
        }
        string ans = "";
        while(!stStr.empty()){
            ans = stStr.top() + ans;
            stStr.pop();
        }
        return ans;
    }
};
